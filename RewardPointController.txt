@RestController
@RequestMapping("/rewardpoints")
public class RewardPointController {
  
    @Autowired
    private RewardPointsRepositoryRepository rewardPointsRepositoryRepository;


GetMapping("/rewardpoints/{customerId}")
public ResponseEntity<List<RewardPoints>> getRewardPoints(@PathVariable int customerId) {
    return ResponseEntity.ok(rewardPointsRepository.findByCustomerId(customerId));
}
